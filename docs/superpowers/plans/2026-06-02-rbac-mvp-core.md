# RBAC MVP Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `@nestarc/rbac` Milestone 0-2: package skeleton, public interfaces, in-memory RBAC engine, NestJS decorators, resolvers, guard, module, and HTTP E2E coverage.

**Architecture:** The package is a TypeScript NestJS library with a small core authorization engine that is independent of HTTP. NestJS integration layers convert route metadata and request context into `RbacService.can()` calls, while storage adapters implement a shared contract. The MVP ships only the in-memory adapter; Prisma and optional integrations are reserved for later plans.

**Tech Stack:** TypeScript, NestJS, reflect-metadata, tsup, vitest, @nestjs/testing, supertest, ESLint, Prettier.

---

## Scope

This plan implements `docs/spec.md` Milestone 0, Milestone 1, and Milestone 2 only.

Included:

- Package skeleton and build/test/lint scripts.
- Root public exports.
- Interfaces and typed errors.
- Permission matcher and input normalization.
- In-memory storage adapter.
- Storage contract tests for the in-memory adapter.
- `RbacService` authorization and management APIs.
- Decorators and metadata semantics.
- Default HTTP subject, tenant, and resource resolvers.
- `RbacGuard` with stable HTTP error mapping.
- `RbacModule.forRoot()` and `forRootAsync()`.
- NestJS E2E test app.
- README quickstart for the implemented scope.

Excluded from this plan:

- Prisma schema, SQL migration, and `PrismaRbacStorage`.
- `@nestarc/tenancy`, `@nestarc/api-keys`, and `@nestarc/audit-log` helper packages.
- Docs site pages beyond README updates.
- Published package release workflow.

```ts
  RbacSubjectMissingError,
  mapRbacErrorToHttpException,

describe('RBAC errors', () => {
  it('stores stable codes and safe details', () => {
    const error = new RbacPermissionDeniedError({ permission: 'reports.write' });

    expect(error.code).toBe('RBAC_PERMISSION_DENIED');
    expect(error.message).toBe('Permission denied');
    expect(error.details).toEqual({ permission: 'reports.write' });

  });

  it('maps storage errors to InternalServerErrorException', () => {
```

```ts
import type {
  AssignRoleStorageInput,
  DeleteRoleInput,
  FindRoleInput,
  GrantPermissionInput,
  ListBindingsStorageInput,
  ListEffectivePermissionsInput,
  ListEffectiveRolesInput,
  ListRolePermissionsInput,
  ListRolesInput,
  RbacEffectivePermission,
  RbacEffectiveRole,
  RbacRole,
  RbacRoleBinding,
  RbacStorage,
  RevokePermissionInput,
  RevokeRoleStorageInput,
  UpsertRoleInput,
} from '../interfaces';

let sequence = 0;

function nextId(prefix: string): string {
  sequence += 1;
  return `${prefix}_${sequence}`;
}

export class InMemoryRbacStorage implements RbacStorage {
  private readonly roles = new Map<string, RbacRole>();
  private readonly bindings = new Map<string, RbacRoleBinding>();

  async findRole(input: FindRoleInput): Promise<RbacRole | null> {
    return [...this.roles.values()].find((role) => role.tenantId === (input.tenantId ?? null) && role.key === input.key) ?? null;
  }

  async upsertRole(input: UpsertRoleInput): Promise<RbacRole> {
    const existing = await this.findRole({ tenantId: input.tenantId ?? null, key: input.key });
    const role: RbacRole = {
      id: existing?.id ?? nextId('role'),
      tenantId: input.tenantId ?? null,
      key: input.key,
      name: input.name,
      description: input.description,
      isSystem: input.isSystem ?? existing?.isSystem ?? false,
      permissions: input.permissions ? [...new Set(input.permissions)] : (existing?.permissions ?? []),
    };
    this.roles.set(role.id, role);
    return role;
  }

  async deleteRole(input: DeleteRoleInput): Promise<void> {
    this.roles.delete(input.roleId);
  }

  async listRoles(input: ListRolesInput): Promise<RbacRole[]> {
    return [...this.roles.values()]
      .filter((role) => input.tenantId === undefined || role.tenantId === (input.tenantId ?? null))
      .sort((a, b) => a.key.localeCompare(b.key));
  }

  async grantPermission(input: GrantPermissionInput): Promise<void> {
    const role = this.roles.get(input.roleId);
    if (!role) return;
    role.permissions = [...new Set([...role.permissions, input.permission])];
  }

  async revokePermission(input: RevokePermissionInput): Promise<void> {
    const role = this.roles.get(input.roleId);
    if (!role) return;
    role.permissions = role.permissions.filter((permission) => permission !== input.permission);
  }

  async listRolePermissions(input: ListRolePermissionsInput): Promise<string[]> {
    return [...(this.roles.get(input.roleId)?.permissions ?? [])].sort();
  }

  async assignRole(input: AssignRoleStorageInput): Promise<RbacRoleBinding> {
    const tenantId = input.tenantId ?? null;
    const resourceType = input.resource?.type ?? null;
    const resourceId = input.resource?.id ?? null;
    const existing = [...this.bindings.values()].find(
      (binding) =>
        binding.revokedAt == null &&
        binding.tenantId === tenantId &&
        binding.subjectType === input.subject.type &&
        binding.subjectId === input.subject.id &&
        binding.roleId === input.roleId &&
        binding.resourceType === resourceType &&
        binding.resourceId === resourceId,
    );
    if (existing) return existing;

    const binding: RbacRoleBinding = {
      id: nextId('binding'),
      tenantId,
      subjectType: input.subject.type,
      subjectId: input.subject.id,
      roleId: input.roleId,
      resourceType,
      resourceId,
      expiresAt: input.expiresAt ?? null,
      revokedAt: null,
      metadata: input.metadata,
    };
    this.bindings.set(binding.id, binding);
    return binding;
  }

  async revokeRole(input: RevokeRoleStorageInput): Promise<void> {
    const binding = this.bindings.get(input.bindingId);
    if (binding && !binding.revokedAt) {
      binding.revokedAt = input.revokedAt ?? new Date();
    }
  }

  async listBindings(input: ListBindingsStorageInput): Promise<RbacRoleBinding[]> {
    return [...this.bindings.values()].filter(
      (binding) =>
        binding.subjectType === input.subject.type &&
        binding.subjectId === input.subject.id &&
        (input.tenantId === undefined || binding.tenantId === (input.tenantId ?? null)),
    );
  }

  async listEffectiveRoles(input: ListEffectiveRolesInput): Promise<RbacEffectiveRole[]> {
    return this.activeBindings(input).flatMap((binding) => {
      const role = this.roles.get(binding.roleId);
      if (!role) return [];
      return [{
        roleKey: role.key,
        roleId: role.id,
        bindingId: binding.id,
        tenantId: binding.tenantId,
        resourceType: binding.resourceType,
        resourceId: binding.resourceId,
        expiresAt: binding.expiresAt,
      }];
    });
  }

  async listEffectivePermissions(input: ListEffectivePermissionsInput): Promise<RbacEffectivePermission[]> {
    const roles = await this.listEffectiveRoles(input);
    return roles.flatMap((effectiveRole) => {
      const role = this.roles.get(effectiveRole.roleId);
      return (role?.permissions ?? []).map((permission) => ({
        ...effectiveRole,
        permission,
      }));
    });
  }

  private activeBindings(input: ListEffectiveRolesInput): RbacRoleBinding[] {
    const now = input.now ?? new Date();
    return [...this.bindings.values()].filter((binding) => {
      if (binding.subjectType !== input.subject.type || binding.subjectId !== input.subject.id) return false;
      if (binding.tenantId !== (input.tenantId ?? null)) return false;
      if (binding.revokedAt) return false;
      if (binding.expiresAt && binding.expiresAt.getTime() < now.getTime()) return false;
      if (!input.resource) return !binding.resourceType && !binding.resourceId;
      if (!binding.resourceType && !binding.resourceId) return true;
      return binding.resourceType === input.resource.type && binding.resourceId === input.resource.id;
    });
  }
}
```

```ts
import {
  RbacPermissionDeniedError,
  RbacStorageError,
  RbacSubjectMissingError,
  RbacTenantMissingError,
} from './errors';
import type {
  AssignRoleInput,
  CreateRoleInput,
  DeleteRoleInput,
  GrantPermissionInput,
  ListBindingsInput,
  ListPermissionsInput,
  ListRolesInput,
  RbacCanInput,
  RbacDecision,
  RbacModuleOptions,
  RbacRole,
  RbacRoleBinding,
  RevokePermissionInput,
  RevokeRoleInput,
  UpdateRoleInput,
} from './interfaces';
import { matchesPermission, normalizePermissions } from './utils';

export class RbacService {
  constructor(private readonly options: RbacModuleOptions) {}

  async can(input: RbacCanInput): Promise<RbacDecision> {
    if (!input.subject?.type || !input.subject.id) {
      return { allowed: false, reason: 'denied_subject_missing' };
    }

    const tenantMode =
      input.tenantMode ?? (this.options.tenant?.requiredByDefault ? 'required' : 'optional');
    const tenantId = tenantMode === 'none' ? null : (input.tenantId ?? input.subject.tenantId ?? null);

    if (tenantMode === 'required' && !tenantId) {
      return {
        allowed: false,
        reason: 'denied_tenant_missing',
        subject: input.subject,
        tenantId,
      };
    }

    if (input.roleKey) {
      return this.canRole(input, tenantId);
    }

    const permissions = normalizePermissions([
      ...(input.permission ? [input.permission] : []),
      ...(input.permissions ?? []),
    ]);
    const mode = input.mode ?? (permissions.length > 1 ? 'all' : 'any');

    try {
      const effective = await this.options.storage.listEffectivePermissions({
        subject: input.subject,
        tenantId,
        resource: input.resource,
        now: input.now ?? this.options.now?.() ?? new Date(),
      });
      const matched = permissions.filter((required) =>
        effective.some((granted) => matchesPermission(granted.permission, required)),
      );
      const allowed = mode === 'all' ? matched.length === permissions.length : matched.length > 0;
      return {
        allowed,
        reason: allowed ? 'allowed_by_role_permission' : 'denied_no_matching_permission',
        subject: input.subject,
        tenantId,
        permission: input.permission,
        permissions,
        mode,
        matchedPermissions: matched,
        matchedRoleKeys: [...new Set(effective.map((entry) => entry.roleKey))],
        resource: input.resource,
      };
    } catch (error) {
      if (this.options.storageErrors === 'throw') throw new RbacStorageError({ cause: String(error) });
      return { allowed: false, reason: 'denied_storage_error', subject: input.subject, tenantId };
    }
  }

  async assertCan(input: RbacCanInput): Promise<void> {
    const decision = await this.can(input);
    if (!decision.allowed) {
      if (decision.reason === 'denied_subject_missing') throw new RbacSubjectMissingError();
      if (decision.reason === 'denied_tenant_missing') throw new RbacTenantMissingError();
      throw new RbacPermissionDeniedError({
        permission: decision.permission,
        permissions: decision.permissions,
        roleKey: decision.roleKey,
      });
    }
  }

  private async canRole(input: RbacCanInput, tenantId: string | null): Promise<RbacDecision> {
    const effective = await this.options.storage.listEffectiveRoles({
      subject: input.subject!,
      tenantId,
      resource: input.resource,
      now: input.now ?? this.options.now?.() ?? new Date(),
    });
    const allowed = effective.some((role) => role.roleKey === input.roleKey);
    return {
      allowed,
      reason: allowed ? 'allowed_by_role' : 'denied_no_matching_role',
      subject: input.subject,
      tenantId,
```

```ts
import type { ExecutionContext } from '@nestjs/common';
import type { RbacSubject, RbacSubjectResolver } from '../interfaces';

function objectValue(value: unknown): Record<string, unknown> | undefined {
  return value && typeof value === 'object' ? (value as Record<string, unknown>) : undefined;
}

function mapUserToSubject(user: unknown): RbacSubject | undefined {
  const record = objectValue(user);
  if (!record) return undefined;
  const id = record.id ?? record.sub ?? record.userId;
  if (!id) return undefined;
  return {
    type: record.type ? String(record.type) : 'user',
    id: String(id),
    tenantId: record.tenantId ? String(record.tenantId) : undefined,
    attributes: record,
  };
}

function mapApiKeyToSubject(value: unknown): RbacSubject | undefined {
  const record = objectValue(value);
  if (!record) return undefined;
  const id = record.keyId ?? record.id;
  if (!id) return undefined;
  return {
    type: 'api_key',
    id: String(id),
    tenantId: record.tenantId ? String(record.tenantId) : undefined,
    attributes: record,
  };
}

export function defaultHttpSubjectResolver(): RbacSubjectResolver {
  return async (context: ExecutionContext) => {
    const request = context.switchToHttp().getRequest<Record<string, unknown>>();
    return (
      mapUserToSubject(request.rbacSubject) ??
      mapUserToSubject(request.user) ??
      mapApiKeyToSubject(request.apiKeyContext) ??
      mapApiKeyToSubject(request.apiKey)
    );
  };
}
```

```ts
import {
  CanActivate,
  ExecutionContext,
  ForbiddenException,
  Inject,
  Injectable,
  UnauthorizedException,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import {
  RBAC_OPTIONS,
  RBAC_REQUIREMENTS_METADATA,
  RBAC_SKIP_METADATA,
  RBAC_SUBJECT_REQUEST_KEY,
} from './constants';
import { RbacResourceMissingError, mapRbacErrorToHttpException } from './errors';
import type { RbacModuleOptions, RbacRequirement, RbacRequirementOptions, RbacSubject } from './interfaces';
import { defaultHttpSubjectResolver, resolveHttpResource, resolveHttpTenant } from './resolvers';
import { RbacService } from './rbac.service';

@Injectable()
export class RbacGuard implements CanActivate {
  constructor(
    private readonly reflector: Reflector,
    private readonly rbac: RbacService,
    @Inject(RBAC_OPTIONS) private readonly options: RbacModuleOptions,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const skip = this.reflector.getAllAndOverride<{ reason?: string }>(RBAC_SKIP_METADATA, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (skip) return true;

    const requirements = this.reflector.getAllAndMerge<RbacRequirement[]>(
      RBAC_REQUIREMENTS_METADATA,
      [context.getClass(), context.getHandler()],
    );
    if (requirements.length === 0) {
      if (this.options.requireMetadata) {
        throw new ForbiddenException({ message: 'Permission denied', code: 'RBAC_PERMISSION_DENIED' });
      }
      return true;
    }

    const subjectResolver = this.options.subjectResolver ?? defaultHttpSubjectResolver();
    const subject = await subjectResolver(context);
    if (!subject) {
      throw new UnauthorizedException({ message: 'Subject missing', code: 'RBAC_SUBJECT_MISSING' });
    }

    const request = context.switchToHttp().getRequest<Record<string, unknown>>();
    request[RBAC_SUBJECT_REQUEST_KEY] = subject;

    for (const requirement of requirements) {
      const requirementOptions = requirement.options ?? {};
      const tenantId = await this.resolveTenant(context, requirementOptions, subject);
      const resource = await this.resolveResource(context, requirementOptions);
      const tenantMode =
        requirementOptions.tenant ??
        (this.options.tenant?.requiredByDefault ? 'required' : 'optional');

      const decision = await this.rbac.can(
        requirement.kind === 'role'
          ? { subject, tenantId, tenantMode, roleKey: requirement.roleKey, resource }
          : {
              subject,
              tenantId,
              tenantMode,
              permissions: requirement.permissions,
              mode: requirement.mode,
              resource,
            },
      );

      if (!decision.allowed) {
        if (decision.reason === 'denied_tenant_missing') {
          throw new ForbiddenException({ message: 'Tenant missing', code: 'RBAC_TENANT_MISSING' });
        }
        throw new ForbiddenException({ message: 'Permission denied', code: 'RBAC_PERMISSION_DENIED' });
      }
    }

    return true;
  }

  private async resolveTenant(
    context: ExecutionContext,
    options: RbacRequirementOptions,
    subject: RbacSubject,
  ) {
    if (this.options.tenantResolver) {
      return this.options.tenantResolver(context, options, subject);
    }
    return resolveHttpTenant(context, options, subject);
  }

  private async resolveResource(context: ExecutionContext, options: RbacRequirementOptions) {
    if (!options.resource) return undefined;
    if (typeof options.resource === 'function') {
      const resolver = new options.resource();
      const resource = await resolver.resolve(context);
      if (!resource) throw mapRbacErrorToHttpException(new RbacResourceMissingError());
      return resource;
    }
    const resource = resolveHttpResource(context, options.resource);
    if (!resource) throw mapRbacErrorToHttpException(new RbacResourceMissingError());
    return resource;
  }
}
```
